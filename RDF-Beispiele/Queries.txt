# Finde alle Ortsteile im Leipziger Osten

PREFIX ld: <http://leipzig-data.de/Data/Model/>
SELECT ?o
from <http://leipzig-data.de/Data/Planungsraeume/>
WHERE {
  <http://leipzig-data.de/Data/Planungsraum/LeipzigerOsten>
    ld:containsOrtsteil ?o .
} 

# Wie viele LD-Adressen im Leipziger Osten? -- 9882

from <http://leipzig-data.de/Data/Planungsraeume/>
from <http://leipzig-data.de/Data/Adressen/>

  <http://leipzig-data.de/Data/Planungsraum/LeipzigerOsten>

}

# Wie viele LD-Orte gibt es in den einzelnen Ortsteilen?

from <http://leipzig-data.de/Data/Planungsraeume/>

?ort a ld:Ort; ld:hasAddress ?a .
?a a ld:Adresse ; ld:inOrtsteil ?o .

# Finde alle LD-Events im Leipziger Osten

PREFIX ld: <http://leipzig-data.de/Data/Model/>
PREFIX ical: <http://www.w3.org/2002/12/cal/ical#> 
SELECT distinct ?e ?ort
from <http://leipzig-data.de/Data/Planungsraeume/>
from <http://leipzig-data.de/Data/Adressen/>
from <http://leipzig-data.de/Data/Orte/>
from <http://leipzig-data.de/Data/Events/>
WHERE {
?e a ld:Event ; ical:location ?ort .
  <http://leipzig-data.de/Data/Planungsraum/LeipzigerOsten>
    ld:containsOrtsteil ?o .
?a a ld:Adresse ; ld:inOrtsteil ?o .
?ort a ld:Ort; ld:hasAddress ?a .
}
